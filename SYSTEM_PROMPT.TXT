scenario_identification:
  description: Critical - Before selecting a strategy, identify which scenario hurts timing most.
  detection_matrix:
    - scenario: Wide LUT Cascades
      detection: Analyze `extract_critical_path_cells` output for >3 LUT levels in series
      threshold: ">3 LUTs on path"
    - scenario: High Fanout Nets
      detection: Parse `report_timing` for `fo=N` values
      threshold: "fanout > 100"
    - scenario: Distributed Logic
      detection: `analyze_critical_path_spread` - cells spread across many tiles
      threshold: "avg_distance > 70 tiles"
    - scenario: Control Logic Imbalance
      detection: Check path delays in `report_timing_summary`
      threshold: "max_delay variation > 2x"
    - scenario: Routing Congestion
      detection: `get_tile_info` on critical path tiles - high site usage
      threshold: "utilization > 80%"
  detection_workflow:
    step_1: |
      Critical Path Analysis
      extract_critical_path_cells(num_paths=50) → analyze LUT levels per path
      report_timing_summary → get WNS, TNS, failing endpoints
    step_2: |
      Candidate Detection
      get_critical_high_fanout_nets(min_fanout=100) → list of candidate nets
      analyze_critical_path_spread → calculate geographic spread
    step_3: |
      Scenario Ranking
      Rank scenarios by impact: WNS contribution × path_count
      Focus on scenarios affecting most critical paths first

implementing_custom_optimizations:
  description: When existing tools are insufficient, implement the transformation as a new RapidWright MCP tool.
  workflow:
    - step: Identify the transformation
      description: What design pattern needs to change?
    - step: Check RapidWright ECO classes
      classes:
        - LUTInputConeOpt
          description: Combines chained LUTs
        - FanOutOptimization
          description: Splits high fanout nets
        - PortDirectioning
          description: Adjusts port directions
        - ReplaceFlopASICWithFPGA
          description: ASIC to FPGA flop mapping
    - step: Implement the tool
      location: rapidwright_tools.py
      template: |
        def optimize_<transformation>(parameters):
            if not _initialized: return {"error": "Not initialized"}
            if _current_design is None: return {"error": "No design loaded"}
            try:
                from com.xilinx.rapidwright.eco import <EcoClass>
                # Apply transformation
                return {"status": "success", "metrics": {...}}
            except Exception as e:
                return {"error": str(e)}
    - step: Register in server.py
      location: TOOL_DEFINITIONS list
