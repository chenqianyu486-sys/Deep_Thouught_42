scenario_identification:
  description: Critical - Before selecting a strategy, identify which scenario hurts timing most.
  detection_matrix:
    - scenario: Wide LUT Cascades
      detection: Analyze `extract_critical_path_cells` output for >3 LUT levels in series
      threshold: ">3 LUTs on path"
    - scenario: High Fanout Nets
      detection: Parse `report_timing` for `fo=N` values
      threshold: "fanout > 100"
    - scenario: Distributed Logic
      detection: `analyze_critical_path_spread` - cells spread across many tiles
      threshold: "avg_distance > 70 tiles"
    - scenario: Control Logic Imbalance
      detection: Check path delays in `report_timing_summary`
      threshold: "max_delay variation > 2x"
    - scenario: Routing Congestion
      detection: `get_tile_info` on critical path tiles - high site usage
      threshold: "utilization > 80%"
  detection_workflow:
    step_1: |
      Critical Path Analysis
      extract_critical_path_cells(num_paths=50) → analyze LUT levels per path
      report_timing_summary → get WNS, TNS, failing endpoints
    step_2: |
      Candidate Detection
      get_critical_high_fanout_nets(min_fanout=100) → list of candidate nets
      analyze_critical_path_spread → calculate geographic spread
    step_3: |
      Scenario Ranking
      Rank scenarios by impact: WNS contribution × path_count
      Focus on scenarios affecting most critical paths first

strategy_selection:
  description: Select strategy based on conditions

    - condition: "recommendation == 'PBLOCK'"
      strategy: Strategy_1_PBLOCK_Based_Replacement

      strategy: Strategy_2_Physical_Optimization
    - condition: "High fanout nets present, no spread"
      strategy: Strategy_3_High_Fanout_Net_Optimization

strategies:
  strategy_1:
    name: PBLOCK-Based Re-placement
    trigger: "recommendation: PBLOCK in initial analysis YAML"
    steps:
      - command: report_utilization_for_pblock
        platform: Vivado
        description: Get resource counts
      - command: analyze_fabric_for_pblock
        platform: RapidWright
        parameters:
          resource_targets:
            LUT: 1.5x
            FF: 1.5x
      - command: convert_fabric_region_to_pblock
        platform: RapidWright
        parameters:
          use_clock_regions: false
      - command: run_tcl("place_design -unplace")
        platform: Vivado
      - command: create_and_apply_pblock
        platform: Vivado
        parameters:
          pblock_ranges_format: "SLICE_X...DSP48E2_X...RAMB..."  # NOT CLOCKREGION
      - command: place_design
        platform: Vivado
        parameters: null
      - command: route_design
        platform: Vivado
        parameters: null
      - command: report_timing_summary
        platform: Vivado

  strategy_2:
    name: Physical Optimization
    trigger: "1-2 problematic paths with spread"
    steps:
      - command: phys_opt_design
        platform: Vivado
        description: Run with appropriate directives
      - command: route_design
        platform: Vivado
      - command: report_timing_summary
        platform: Vivado

  strategy_3:
    name: High Fanout Net Optimization
    trigger: "High fanout nets present, no spread"

          overwrite: true
          directory: temp
      - command: open_checkpoint
        platform: Vivado
        note: "NOT input file!"
      - command: route_design
        platform: Vivado
      - command: report_timing_summary

measurement_flow:
  baseline_measurement:
    description: Before any optimization
    commands:
      - report_timing_summary
        output: WNS, TNS, failing_endpoints
      - get_wns
        output: baseline WNS value
  intermediate_measurements:
    description: After each timing-affecting operation
    commands:
      - report_timing_summary
        output: compare WNS improvement
      - get_wns
        output: confirm numeric change
  final_verification:
    description: After optimization completes
    commands:
      - report_timing_summary
        output: full timing report
      - report_route_status
        output: routing completion
  impact_assessment:
    format: YAML
    fields:
      - baseline_wns: "<value before optimization>"
      - final_wns: "<value after optimization>"
      - improvement: "<final_wns - baseline_wns>"
      - status: IMPROVED | REGRESSION | UNCHANGED

common_failure_patterns:
  - pattern: "route_design times out after fanout optimization"
    cause: "Too many nets unplaced, causing routing congestion"
    fix: "Run phys_opt_design before routing"

  - pattern: "WNS gets worse after optimize_fanout"
    cause: "Split_factor too aggressive, new cells introduced routing delays"
    fix: "Reduce split_factor by 50% and retry"

  - pattern: "get_wns returns PARSE_ERROR"
    cause: "Timing report format unrecognized by parser"
    fix: "Use report_timing_summary instead for WNS extraction"

  - pattern: "PBLOCK too small, place_design fails"
    cause: "PBLOCK doesn't have enough resources for design"
    fix: "Use analyze_fabric_for_pblock to size PBLOCK correctly (1.5x resource multiplier)"

  - pattern: "No design open error in Vivado"
    cause: "Vivado state lost after crash or disconnect"
    fix: "Re-open best checkpoint DCP and continue"
